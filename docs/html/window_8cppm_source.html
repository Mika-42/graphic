73 std::vector<View*>& getHoveredViews(const MouseEventView& mouse) noexcept {
74 hovered.clear();
75 std::function<void(View*)> collect = [&](View* node) noexcept {
76 if (!node) return;
77
78 for (auto& child : node->getChildren()) {
79
80 if (!child || !child->isVisible()) {
81 continue;
82 }
83
84 if (child->isClip() && !child->clipContain(mouse)) {
85 continue;
86 }
87
88 if (child->contain(mouse)) {
89 hovered.push_back(child.get());
90 collect(child.get());
91 }
92 }
93 };
94
95 collect(children[0].get());
96 return hovered; // Top-most = hovered.back() !
97 }

104 if (!view || !view->isVisible()) return;
105
106 // Trier les enfants par zIndex

109 // Parcours récursif des enfants
