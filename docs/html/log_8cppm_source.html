370 template <typename... Args>
371 static void print_colored(Level lvl, std::source_location loc, std::string_view fmt, Args &&...args) {
372 std::string_view color = colors[static_cast<size_t>(lvl)];
373
374 std::print(std::cerr, "{}{}[{}] ", bold, color, level_name(lvl), reset);
375
376 if constexpr (sizeof...(args) > 0) {
377 std::print(std::cerr, "{}", std::vformat(fmt, std::make_format_args(args...)));
378 } else {
379 std::print(std::cerr, "{}", fmt);
380 }
381
382 if (loc.file_name() && *loc.file_name() != '\0') {
383 std::print(std::cerr, " {:<20}:{:<4} {:<30} |",
384 std::filesystem::path(loc.file_name()).filename().string(), loc.line(),
385 std::string_view(loc.function_name()).substr(0, 30));
386 }
387
388 std::print(std::cerr, "{}\n", reset);
389 }
